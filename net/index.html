Three languages. One API. Zero excuses.

# Generate code with Code Pilot
curl -X POST https://api.serlf.net/v1/codepilot/generate \
  -H "Authorization: Bearer $SERLF_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "prompt": "Build a REST API for a todo app",
    "lang": "python"
  }'
import serlf

client = serlf.Client(api_key="your-key")

# Generate code with Code Pilot
result = client.codepilot.generate(
  prompt="Build a REST API for a todo app",
  lang="python"
)

print(result.code)
import { Serlf } from '@serlf/sdk';

const client = new Serlf('your-key');

// Generate code with Code Pilot
const result = await client.codepilot.generate({
  prompt: 'Build a REST API for a todo app',
  lang: 'python'
});

console.log(result.code);
